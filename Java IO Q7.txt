import java.util.Scanner;
class Main{
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int num = sc.nextInt();
        String hex = Integer.toHexString(num); 
        System.out.println("Hexadecimal (lowercase): " + hex);
        System.out.println("Hexadecimal (uppercase): " + hex.toUpperCase());
        sc.close();
    }
}
